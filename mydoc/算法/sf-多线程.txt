4.生产者 消费者
public class All {
    private static final int SLEEPTIME = 10000;
    public static class PCData1 {
        private final int intData;
        public PCData1(int d) {
            intData = d;
        }
        public int getIntData() {
            return intData;
        }
    }
    public static class Consumer1 implements Runnable {
        private BlockingQueue<PCData1> queue;
        public Consumer1(BlockingQueue<PCData1> queue) {
            this.queue = queue;
        }
        @Override
        public void run() {
            Random r = new Random();
            long threadId = Thread.currentThread().getId();
            while (true) {
                PCData1 data1 = null;
                try {
                    Thread.sleep(r.nextInt(SLEEPTIME));
                    data1 = queue.take();
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
                if (data1 != null) {
                    int result = data1.intData * data1.intData;
                    System.out.println("消费者：" + threadId + "将data :" + data1.intData + "消费" + MessageFormat.format("{0}*{1}={2}", data1.intData, data1.intData, result));
                }
            }
        }
    }
    public static class Producer1 implements Runnable {
        private BlockingQueue<PCData1> queue;
        private static AtomicInteger num = new AtomicInteger(1);
        public Producer1(BlockingQueue<PCData1> queue) {
            this.queue = queue;
        }
        @Override
        public void run() {
            long threadId = Thread.currentThread().getId();
            System.out.println("start producer " + threadId);
            Random r = new Random();
            while (true) {
                PCData1 data1;
                data1 = new PCData1(num.incrementAndGet());
                try {
                    queue.offer(data1, 2, TimeUnit.SECONDS);
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
                System.out.println("生产者：" + threadId + "将data :" + data1.intData + "加入队列");
                try {
                    Thread.sleep(r.nextInt(SLEEPTIME));
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
        }
    }

    public static void main(String[] args) {
        BlockingQueue<PCData1> queue = new LinkedBlockingQueue<>(10);
        Producer1 p1 = new Producer1(queue);
        Producer1 p2 = new Producer1(queue);
        Producer1 p3 = new Producer1(queue);
        Consumer1 c1 = new Consumer1(queue);
        Consumer1 c2 = new Consumer1(queue);
        Consumer1 c3 = new Consumer1(queue);
        ExecutorService service = Executors.newCachedThreadPool();
        service.execute(p1);
        service.execute(p2);
        service.execute(p3);
        service.execute(c1);
        service.execute(c2);
        service.execute(c3);
    }

}
